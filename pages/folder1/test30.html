# 1、获取用户数据，插入到数据库中

在页面上设置点击一个【获取用户信息】的按钮，通过按钮触发点击事件函数，这个事件函数内部有一个调用云函数的回调函数， 通过这个回调函数，我们可以到【云函数的列表】下面找到我们在回调函数中选定的【name：云函数名】， 然后通过这个云函数的 index.js 异步获取用户的信息

const cloud = require('wx-server-sdk')

//初始化
cloud.init()

// 云函数入口函数
//一个云函数就是一个项目，独立的 serverless
//main就是入口
exports.main = async (event, context) => {
  //event是一个请求事件
  return event.userInfo
}

再回到 页面的index.js里面，当回调函数调用了云函数后,可以得到返回结果：

# 2、进行文件/照片上传，并把文件/照片插入到云数据库中

在前面的步骤中，我们在获取用户信息成功后会跳转到一个【上传文件】的界面

先创建集合photos

点击后触发事件bindtap="upLoad"，然后来看add.js

upLoad() {
    //上传图片，或者拍照
    wx.chooseImage({
      count:9,
      // 压缩与否 
      sizeType: ['original', 'compressed'],
      // 相册， 拍摄
      sourceType: ['album', 'camera'],
      success: res => {
        // 拿到图片文件 
        console.log(res);
        const tempFilePaths = res.tempFilePaths
        //每张图片单独做
        for(var i = 0; i < tempFilePaths.length; i ++ ){
          //生成图片的文件名，产生一个唯一的随机数
          let randString = Math.floor(Math.random() * 1000000).toString() + '.png'
          //向云端上传文件
          wx.cloud.uploadFile({
            cloudPath: randString, 
            // 上传至云端的路径 
            //目的地址
            filePath: tempFilePaths[i], 
            // 小程序临时文件路径 
            //文件地址
            success: res => {
              //console.log(res)
              photos.add({
                data: {
                  image: res.fileID
                }
              }).then(res => {
                wx.showToast({
                  title: '上传成功',
                  icon: 'success'
                })
              })
            },
            fail: console.error
          })
        }
      
      },
      fail: err => {
        console.error(err)
      }
    })

  }
